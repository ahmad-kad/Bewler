54 class ExtendedKalmanFilter:
55  """
56  Extended Kalman Filter for sensor fusion.
57 
58  Fuses:
59  - SLAM pose (position + orientation)
60  - GPS position (lat/lon → local)
61  - IMU heading (optional yaw correction)
62 
63  State: [x, y, z, yaw, vx, vy]
64  """
65 
66  def __init__(self, state_dim: int = 6, measurement_dim: int = 3):
67  """
68  Initialize EKF.
69 
70  Args:
71  state_dim: Dimension of state vector (x, y, z, yaw, vx, vy)
72  measurement_dim: Dimension of measurement (position x, y, yaw)
73  """
74  self.statestate = np.zeros(state_dim)
75  self.PP = np.eye(state_dim) * 0.1 # Initial covariance (high uncertainty)
76 
77  # Process noise covariance (how much we trust the motion model)
78  self.QQ = np.eye(state_dim)
79  self.QQ[:2, :2] *= 0.01 # Position process noise
80  self.QQ[2, 2] = 0.01 # Z process noise
81  self.QQ[3, 3] = 0.01 # Yaw process noise
82  self.QQ[4:, 4:] *= 0.1 # Velocity process noise
83 
84  # Measurement noise covariance (how much we trust measurements)
85  self.R_slamR_slam = np.eye(3) * 0.05 # SLAM measurement noise
86  self.R_gpsR_gps = np.eye(2) * 1.0 # GPS measurement noise (lower confidence)
87 
88  self.dtdt = 0.1 # Time step
89  self.gps_referencegps_reference: Optional[Tuple[float, float]] = None

91  def predict(self, dt: float) -> None:
92  """
93  Predict step: project state forward in time.
94 
95  Motion model assumes constant velocity.
96  """
97  self.dtdt = dt
98 
99  # State transition matrix (constant velocity model)
100  F = np.eye(6)
101  F[0, 4] = dt # x += vx * dt
102  F[1, 5] = dt # y += vy * dt
103 
104  # Update state: x = F * x
105  self.statestate = F @ self.statestate
106 
107  # Update covariance: P = F * P * F^T + Q
108  self.PP = F @ self.PP @ F.T + self.QQ
109 
110  def update_slam(self, position: np.ndarray, orientation_yaw: float,
111  covariance: Optional[np.ndarray] = None) -> None:
112  """
113  Update with SLAM measurement.
114 
115  Args:
116  position: [x, y, z] position from SLAM
117  orientation_yaw: Yaw angle from SLAM
118  covariance: Optional 3x3 covariance matrix
119  """
120  # Measurement: [x, y, yaw]
121  z = np.array([position[0], position[1], orientation_yaw])
122 
123  # Measurement Jacobian (linear in this case)
124  H = np.zeros((3, 6))
125  H[0, 0] = 1.0 # Measure x
126  H[1, 1] = 1.0 # Measure y
127  H[2, 3] = 1.0 # Measure yaw
128 
129  # Use provided covariance or default
130  if covariance is not None:
131  R = covariance[:3, :3]
132  else:
133  R = self.R_slamR_slam
134 
135  # Innovation (measurement residual)
136  y = z - (H @ self.statestate)
137 
138  # Normalize angle difference
139  y[2] = self._normalize_angle_normalize_angle(y[2])
140 
141  # Innovation covariance: S = H * P * H^T + R
142  S = H @ self.PP @ H.T + R
143 
144  # Kalman gain: K = P * H^T * S^{-1}
145  K = self.PP @ H.T @ np.linalg.inv(S)
146 
147  # Update state: x = x + K * y
148  self.statestate = self.statestate + K @ y
149 
150  # Update covariance: P = (I - K * H) * P
151  I = np.eye(6)
152  self.PP = (I - K @ H) @ self.PP
