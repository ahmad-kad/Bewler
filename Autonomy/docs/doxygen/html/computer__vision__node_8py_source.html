230  def rotation_matrix_to_quaternion(self, R: np.ndarray) -> Tuple[float, float, float, float]:
231  """Convert rotation matrix to quaternion."""
232  q = np.zeros(4)
233  trace = np.trace(R)
234  if trace > 0:
235  s = 0.5 / math.sqrt(trace + 1.0)
236  q[3] = 0.25 / s
237  q[0] = (R[2, 1] - R[1, 2]) * s
238  q[1] = (R[0, 2] - R[2, 0]) * s
239  q[2] = (R[1, 0] - R[0, 1]) * s
240  else:
241  if R[0, 0] > R[1, 1] and R[0, 0] > R[2, 2]:
242  s = 2.0 * math.sqrt(1.0 + R[0, 0] - R[1, 1] - R[2, 2])
243  q[3] = (R[2, 1] - R[1, 2]) / s
244  q[0] = 0.25 * s
245  q[1] = (R[0, 1] + R[1, 0]) / s
246  q[2] = (R[0, 2] + R[2, 0]) / s
247  elif R[1, 1] > R[2, 2]:
248  s = 2.0 * math.sqrt(1.0 + R[1, 1] - R[0, 0] - R[2, 2])
249  q[3] = (R[0, 2] - R[2, 0]) / s
250  q[0] = (R[0, 1] + R[1, 0]) / s
251  q[1] = 0.25 * s
252  q[2] = (R[1, 2] + R[2, 1]) / s
253  else:
254  s = 2.0 * math.sqrt(1.0 + R[2, 2] - R[0, 0] - R[1, 1])
255  q[3] = (R[1, 0] - R[0, 1]) / s
256  q[0] = (R[0, 2] + R[2, 0]) / s
257  q[1] = (R[1, 2] + R[2, 1]) / s
258  q[2] = 0.25 * s
