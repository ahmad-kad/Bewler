141  def _calculate_plane_normal(self, positions: List[Point]) -> Tuple[np.ndarray, Quaternion]:
142  """Calculate plane normal using Principal Component Analysis."""
143  # Convert to numpy array
144  points = np.array([[pos.x, pos.y, pos.z] for pos in positions])
145 
146  # Center the points

168  def _normal_to_quaternion(self, normal: np.ndarray) -> Quaternion:
169  """Convert normal vector to quaternion orientation."""
170  # Default orientation (facing forward)
171  default_normal = np.array([0, 0, 1])
172 
173  # Calculate rotation axis
174  rotation_axis = np.cross(default_normal, normal)
175  rotation_axis = rotation_axis / np.linalg.norm(rotation_axis)
176 
177  # Calculate rotation angle
178  cos_angle = np.dot(default_normal, normal)
179  angle = math.acos(np.clip(cos_angle, -1.0, 1.0))
180 
181  # Convert to quaternion
182  quat = Quaternion()
183  quat.w = math.cos(angle / 2)
184  quat.x = rotation_axis[0] * math.sin(angle / 2)
185  quat.y = rotation_axis[1] * math.sin(angle / 2)
186  quat.z = rotation_axis[2] * math.sin(angle / 2)
