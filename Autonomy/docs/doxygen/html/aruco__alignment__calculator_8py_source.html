221  def _calculate_plane_normal(self, positions: List[Point]) -> Tuple[np.ndarray, Quaternion]:
222  """Calculate plane normal using Principal Component Analysis."""
223  # Convert to numpy array
224  points = np.array([[pos.x, pos.y, pos.z] for pos in positions])
225 
226  # Center the points
227  centroid = np.mean(points, axis=0)
228  centered_points = points - centroid

230  # Calculate covariance matrix
231  cov_matrix = np.cov(centered_points.T)

233  # Find eigenvalues and eigenvectors

235 
236  # The normal is the eigenvector with smallest eigenvalue

240  if normal[2] > 0:

297  def _calculate_alignment_quality(
298  self,
299  positions: List[Point],
300  center: Point,
301  normal: np.ndarray,
302  config: Dict[str, Any]
303  ) -> float:
304  """Calculate alignment quality score (0.0-1.0)."""
305  if len(positions) < 3:
306  return 0.0
307 
308  # Calculate how well points fit the plane
309  plane_errors = []
310  for pos in positions:
311  # Distance from point to plane
312  point_vec = np.array([pos.x - center.x, pos.y - center.y, pos.z - center.z])
313  distance = abs(np.dot(point_vec, normal))
314  plane_errors.append(distance)
315 
316  # Calculate quality based on plane fit
317  max_error = max(plane_errors)
318  tolerance = config["alignment_tolerance"]
319  quality = max(0.0, 1.0 - (max_error / tolerance))
320 
321  # Penalize for missing tags
322  expected_tags = config["required_tags"]
323  detected_tags = len(positions)
324  tag_penalty = (expected_tags - detected_tags) / expected_tags
325  quality *= (1.0 - tag_penalty * 0.3) # 30% penalty for missing tags
326 
327  return min(1.0, max(0.0, quality))
