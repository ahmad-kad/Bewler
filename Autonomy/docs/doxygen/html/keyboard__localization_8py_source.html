249  def _rotation_matrix_to_quaternion(self, R: np.ndarray) -> np.ndarray:
250  """Convert 3x3 rotation matrix to quaternion [x, y, z, w]."""
251  trace = np.trace(R)
252  q = np.zeros(4)
253 
254  if trace > 0:
255  s = 0.5 / np.sqrt(trace + 1.0)
256  q[3] = 0.25 / s
257  q[0] = (R[2, 1] - R[1, 2]) * s
258  q[1] = (R[0, 2] - R[2, 0]) * s
259  q[2] = (R[1, 0] - R[0, 1]) * s
260  elif R[0, 0] > R[1, 1] and R[0, 0] > R[2, 2]:
261  s = 2.0 * np.sqrt(1.0 + R[0, 0] - R[1, 1] - R[2, 2])
262  q[3] = (R[2, 1] - R[1, 2]) / s
263  q[0] = 0.25 * s
264  q[1] = (R[0, 1] + R[1, 0]) / s
265  q[2] = (R[0, 2] + R[2, 0]) / s
266  elif R[1, 1] > R[2, 2]:
267  s = 2.0 * np.sqrt(1.0 + R[1, 1] - R[0, 0] - R[2, 2])
268  q[3] = (R[0, 2] - R[2, 0]) / s
269  q[0] = (R[0, 1] + R[1, 0]) / s
270  q[1] = 0.25 * s
271  q[2] = (R[1, 2] + R[2, 1]) / s
272  else:
273  s = 2.0 * np.sqrt(1.0 + R[2, 2] - R[0, 0] - R[1, 1])
274  q[3] = (R[1, 0] - R[0, 1]) / s
275  q[0] = (R[0, 2] + R[2, 0]) / s
276  q[1] = (R[1, 2] + R[2, 1]) / s
277  q[2] = 0.25 * s
278 
279  return q
